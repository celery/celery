19class EventDispatcher: 

20 """Dispatches event messages. 

21 

22 Arguments: 

23 connection (kombu.Connection): Connection to the broker. 

24 

25 hostname (str): Hostname to identify ourselves as, 

26 by default uses the hostname returned by 

27 :func:`~celery.utils.anon_nodename`. 

28 

29 groups (Sequence[str]): List of groups to send events for. 

30 :meth:`send` will ignore send requests to groups not in this list. 

31 If this is :const:`None`, all events will be sent. 

32 Example groups include ``"task"`` and ``"worker"``. 

33 

34 enabled (bool): Set to :const:`False` to not actually publish any 

35 events, making :meth:`send` a no-op. 

36 

37 channel (kombu.Channel): Can be used instead of `connection` to specify 

38 an exact channel to use when sending events. 

39 

40 buffer_while_offline (bool): If enabled events will be buffered 

41 while the connection is down. :meth:`flush` must be called 

42 as soon as the connection is re-established. 

43 

44 Note: 

45 You need to :meth:`close` this after use. 

46 """

116 def publish(self, type, fields, producer, 

117 blind=False, Event=Event, **kwargs): 

118 """Publish event using custom :class:`~kombu.Producer`. 

119 

120 Arguments: 

121 type (str): Event type name, with group separated by dash (`-`). 

122 fields: Dictionary of event fields, must be json serializable. 

123 producer (kombu.Producer): Producer instance to use: 

124 only the ``publish`` method will be called. 

125 retry (bool): Retry in the event of connection failure. 

126 retry_policy (Mapping): Map of custom retry policy options. 

127 See :meth:`~kombu.Connection.ensure`. 

128 blind (bool): Don't set logical clock value (also don't forward 

129 the internal logical clock). 

130 Event (Callable): Event type used to create event. 

131 Defaults to :func:`Event`. 

132 utcoffset (Callable): Function returning the current 

133 utc offset in hours. 

134 """ 

135 clock = None if blind else self.clock.forward() 

136 event = Event(type, hostname=self.hostname, utcoffset=utcoffset(), 

137 pid=self.pid, clock=clock, **fields) 

138 with self.mutex: 

139 return self._publish(event, producer, 

140 routing_key=type.replace('-', '.'), **kwargs) 

141 

142 def _publish(self, event, producer, routing_key, retry=False, 

143 retry_policy=None, utcoffset=utcoffset): 

144 exchange = self.exchange 

145 try: 

146 producer.publish( 

147 event, 

148 routing_key=routing_key, 

149 exchange=exchange.name, 

150 retry=retry, 

151 retry_policy=retry_policy, 

152 declare=[exchange], 

153 serializer=self.serializer, 

154 headers=self.headers, 

155 delivery_mode=self.delivery_mode, 

156 ) 

157 except Exception as exc: # pylint: disable=broad-except 

158 if not self.buffer_while_offline: 

159 raise 

160 self._outbound_buffer.append((event, routing_key, exc))

162 def send(self, type, blind=False, utcoffset=utcoffset, retry=False, 

163 retry_policy=None, Event=Event, **fields): 

164 """Send event. 

165 

166 Arguments: 

167 type (str): Event type name, with group separated by dash (`-`). 

168 retry (bool): Retry in the event of connection failure. 

169 retry_policy (Mapping): Map of custom retry policy options. 

170 See :meth:`~kombu.Connection.ensure`. 

171 blind (bool): Don't set logical clock value (also don't forward 

172 the internal logical clock). 

173 Event (Callable): Event type used to create event, 

174 defaults to :func:`Event`. 

175 utcoffset (Callable): unction returning the current utc offset 

176 in hours. 

177 **fields (Any): Event fields -- must be json serializable. 

178 """ 

179 if self.enabled: 

180 groups, group = self.groups, group_from(type) 

181 if groups and group not in groups: 

182 return 

183 if group in self.buffer_group: 

184 clock = self.clock.forward() 

185 event = Event(type, hostname=self.hostname, 

186 utcoffset=utcoffset(), 

187 pid=self.pid, clock=clock, **fields) 

188 buf = self._group_buffer[group] 

189 buf.append(event) 

190 if len(buf) >= self.buffer_limit: 

191 self.flush() 

192 elif self.on_send_buffered: 

193 self.on_send_buffered() 

194 else: 

195 return self.publish(type, fields, self.producer, blind=blind, 

196 Event=Event, retry=retry, 

197 retry_policy=retry_policy)

199 def flush(self, errors=True, groups=True): 

200 """Flush the outbound buffer.""" 

201 if errors: 

202 buf = list(self._outbound_buffer) 

203 try: 

204 with self.mutex: 

205 for event, routing_key, _ in buf: 

206 self._publish(event, self.producer, routing_key) 

207 finally: 

208 self._outbound_buffer.clear() 

209 if groups: 

210 with self.mutex: 

211 for group, events in self._group_buffer.items(): 

212 self._publish(events, self.producer, '%s.multi' % group) 

213 events[:] = [] # list.clear
