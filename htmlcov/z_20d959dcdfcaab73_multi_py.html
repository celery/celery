1"""Start multiple worker instances from the command-line. 

2 

3.. program:: celery multi 

4 

5Examples 

6======== 

7 

8.. code-block:: console 

9 

10 $ # Single worker with explicit name and events enabled. 

11 $ celery multi start Leslie -E 

12 

13 $ # Pidfiles and logfiles are stored in the current directory 

14 $ # by default. Use --pidfile and --logfile argument to change 

15 $ # this. The abbreviation %n will be expanded to the current 

16 $ # node name. 

17 $ celery multi start Leslie -E --pidfile=/var/run/celery/%n.pid 

18 --logfile=/var/log/celery/%n%I.log 

19 

20 

21 $ # You need to add the same arguments when you restart, 

22 $ # as these aren't persisted anywhere. 

23 $ celery multi restart Leslie -E --pidfile=/var/run/celery/%n.pid 

24 --logfile=/var/log/celery/%n%I.log 

25 

26 $ # To stop the node, you need to specify the same pidfile. 

27 $ celery multi stop Leslie --pidfile=/var/run/celery/%n.pid 

28 

29 $ # 3 workers, with 3 processes each 

30 $ celery multi start 3 -c 3 

31 celery worker -n celery1@myhost -c 3 

32 celery worker -n celery2@myhost -c 3 

33 celery worker -n celery3@myhost -c 3 

34 

35 $ # override name prefix when using range 

36 $ celery multi start 3 --range-prefix=worker -c 3 

37 celery worker -n worker1@myhost -c 3 

38 celery worker -n worker2@myhost -c 3 

39 celery worker -n worker3@myhost -c 3 

40 

41 $ # start 3 named workers 

42 $ celery multi start image video data -c 3 

43 celery worker -n image@myhost -c 3 

44 celery worker -n video@myhost -c 3 

45 celery worker -n data@myhost -c 3 

46 

47 $ # specify custom hostname 

48 $ celery multi start 2 --hostname=worker.example.com -c 3 

49 celery worker -n celery1@worker.example.com -c 3 

50 celery worker -n celery2@worker.example.com -c 3 

51 

52 $ # specify fully qualified nodenames 

53 $ celery multi start foo@worker.example.com bar@worker.example.com -c 3 

54 

55 $ # fully qualified nodenames but using the current hostname 

56 $ celery multi start foo@%h bar@%h 

57 

58 $ # Advanced example starting 10 workers in the background: 

59 $ # * Three of the workers processes the images and video queue 

60 $ # * Two of the workers processes the data queue with loglevel DEBUG 

61 $ # * the rest processes the default' queue. 

62 $ celery multi start 10 -l INFO -Q:1-3 images,video -Q:4,5 data 

63 -Q default -L:4,5 DEBUG 

64 

65 $ # You can show the commands necessary to start the workers with 

66 $ # the 'show' command: 

67 $ celery multi show 10 -l INFO -Q:1-3 images,video -Q:4,5 data 

68 -Q default -L:4,5 DEBUG 

69 

70 $ # Additional options are added to each celery worker's command, 

71 $ # but you can also modify the options for ranges of, or specific workers 

72 

73 $ # 3 workers: Two with 3 processes, and one with 10 processes. 

74 $ celery multi start 3 -c 3 -c:1 10 

75 celery worker -n celery1@myhost -c 10 

76 celery worker -n celery2@myhost -c 3 

77 celery worker -n celery3@myhost -c 3 

78 

79 $ # can also specify options for named workers 

80 $ celery multi start image video data -c 3 -c:image 10 

81 celery worker -n image@myhost -c 10 

82 celery worker -n video@myhost -c 3 

83 celery worker -n data@myhost -c 3 

84 

85 $ # ranges and lists of workers in options is also allowed: 

86 $ # (-c:1-3 can also be written as -c:1,2,3) 

87 $ celery multi start 5 -c 3 -c:1-3 10 

88 celery worker -n celery1@myhost -c 10 

89 celery worker -n celery2@myhost -c 10 

90 celery worker -n celery3@myhost -c 10 

91 celery worker -n celery4@myhost -c 3 

92 celery worker -n celery5@myhost -c 3 

93 

94 $ # lists also works with named workers 

95 $ celery multi start foo bar baz xuzzy -c 3 -c:foo,bar,baz 10 

96 celery worker -n foo@myhost -c 10 

97 celery worker -n bar@myhost -c 10 

98 celery worker -n baz@myhost -c 10 

99 celery worker -n xuzzy@myhost -c 3

262 def execute_from_commandline(self, argv, cmd=None): 

263 # Reserve the --nosplash|--quiet|-q/--verbose options. 

264 argv = self._handle_reserved_options(argv) 

265 self.cmd = cmd if cmd is not None else self.cmd 

266 self.prog_name = os.path.basename(argv.pop(0)) 

267 

268 if not self.validate_arguments(argv): 

269 return self.error() 

270 

271 return self.call_command(argv[0], argv[1:]) 

272 

273 def validate_arguments(self, argv): 

274 return argv and argv[0][0] != '-' 

275 

276 def call_command(self, command, argv): 

277 try: 

278 return self.commands[command](*argv) or EX_OK 

279 except KeyError: 

280 return self.error(f'Invalid command: {command}') 

281 

282 def _handle_reserved_options(self, argv): 

283 argv = list(argv) # don't modify callers argv. 

284 for arg, attr in self.reserved_options: 

285 if arg in argv: 

286 setattr(self, attr, bool(argv.pop(argv.index(arg)))) 

287 return argv

344 def _find_sig_argument(self, p, default=signal.SIGTERM): 

345 args = p.args[len(p.values):] 

346 for arg in reversed(args): 

347 if len(arg) == 2 and arg[0] == '-': 

348 try: 

349 return int(arg[1]) 

350 except ValueError: 

351 pass 

352 if arg[0] == '-': 

353 try: 

354 return signals.signum(arg[1:]) 

355 except (AttributeError, TypeError): 

356 pass 

357 return default

359 def _nodes_from_argv(self, argv, cmd=None): 

360 cmd = cmd if cmd is not None else self.cmd 

361 p = self.OptionParser(argv) 

362 p.parse() 

363 return p, self.MultiParser(cmd=cmd).parse(p)

472def multi(ctx, **kwargs): 

473 """Start multiple worker instances.""" 

474 cmd = MultiTool(quiet=ctx.obj.quiet, no_color=ctx.obj.no_color) 

475 # In 4.x, celery multi ignores the global --app option. 

476 # Since in 5.0 the --app option is global only we 

477 # rearrange the arguments so that the MultiTool will parse them correctly. 

478 args = sys.argv[1:] 

479 args = args[args.index('multi'):] + args[:args.index('multi')] 

480 return cmd.execute_from_commandline(args)
