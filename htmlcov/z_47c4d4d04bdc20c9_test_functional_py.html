175 self.iter_ = iter_ 

176 self._errored = False 

177 

178 def __iter__(self): 

179 yield from self.iter_ 

180 if not self._errored: 

181 try: 

182 # This should stop the regen instance from marking

184 raise AssertionError("Iterator errored")

186 self._errored = True

188 original_list = list(range(42))

203 next(iter_g)
