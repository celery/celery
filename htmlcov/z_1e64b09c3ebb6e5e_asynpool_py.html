141def _select(readers=None, writers=None, err=None, timeout=0, 

142 poll=_select_imp): 

143 """Simple wrapper to :class:`~select.select`, using :`~select.poll`. 

144 

145 Arguments: 

146 readers (Set[Fd]): Set of reader fds to test if readable. 

147 writers (Set[Fd]): Set of writer fds to test if writable. 

148 err (Set[Fd]): Set of fds to test for error condition. 

149 

150 All fd sets passed must be mutable as this function 

151 will remove non-working fds from them, this also means 

152 the caller must make sure there are still fds in the sets 

153 before calling us again. 

154 

155 Returns: 

156 Tuple[Set, Set, Set]: of ``(readable, writable, again)``, where 

157 ``readable`` is a set of fds that have data available for read, 

158 ``writable`` is a set of fds that's ready to be written to 

159 and ``again`` is a flag that if set means the caller must 

160 throw away the result and call us again. 

161 """ 

162 readers = set() if readers is None else readers 

163 writers = set() if writers is None else writers 

164 err = set() if err is None else err 

165 try: 

166 return poll(readers, writers, err, timeout) 

167 except OSError as exc: 

168 _errno = exc.errno 

169 

170 if _errno == errno.EINTR: 

171 return set(), set(), 1 

172 elif _errno in SELECT_BAD_FD: 

173 for fd in readers | writers | err: 

174 try: 

175 select.select([fd], [], [], 0) 

176 except OSError as exc: 

177 _errno = exc.errno 

178 

179 if _errno not in SELECT_BAD_FD: 

180 raise 

181 readers.discard(fd) 

182 writers.discard(fd) 

183 err.discard(fd)

189def iterate_file_descriptors_safely(fds_iter, source_data, 

190 hub_method, *args, **kwargs): 

191 """Apply hub method to fds in iter, remove from list if failure. 

192 

193 Some file descriptors may become stale through OS reasons 

194 or possibly other reasons, so safely manage our lists of FDs. 

195 :param fds_iter: the file descriptors to iterate and apply hub_method 

196 :param source_data: data source to remove FD if it renders OSError 

197 :param hub_method: the method to call with each fd and kwargs 

198 :*args to pass through to the hub_method; 

199 with a special syntax string '*fd*' represents a substitution 

200 for the current fd object in the iteration (for some callers). 

201 :**kwargs to pass through to the hub method (no substitutions needed) 

202 """ 

203 def _meta_fd_argument_maker(): 

204 # uses the current iterations value for fd 

205 call_args = args 

206 if "*fd*" in call_args: 

207 call_args = [fd if arg == "*fd*" else arg for arg in args] 

208 return call_args 

209 # Track stale FDs for cleanup possibility 

210 stale_fds = [] 

211 for fd in fds_iter: 

212 # Handle using the correct arguments to the hub method 

213 hub_args, hub_kwargs = _meta_fd_argument_maker(), kwargs 

214 try: # Call the hub method 

215 hub_method(fd, *hub_args, **hub_kwargs) 

216 except (OSError, FileNotFoundError): 

217 logger.warning( 

218 "Encountered OSError when accessing fd %s ", 

219 fd, exc_info=True) 

220 stale_fds.append(fd) # take note of stale fd 

221 # Remove now defunct fds from the managed list 

222 if source_data: 

223 for fd in stale_fds: 

224 try: 

225 if hasattr(source_data, 'remove'): 

226 source_data.remove(fd) 

227 else: # then not a list/set ... try dict 

228 source_data.pop(fd, None) 

229 except ValueError: 

230 logger.warning("ValueError trying to invalidate %s from %s", 

231 fd, source_data)

792 def schedule_writes(ready_fds, total_write_count=None): 

793 if not total_write_count: 

794 total_write_count = [0] 

795 # Schedule write operation to ready file descriptor. 

796 # The file descriptor is writable, but that does not 

797 # mean the process is currently reading from the socket. 

798 # The socket is buffered so writable simply means that 

799 # the buffer can accept at least 1 byte of data. 

800 

801 # This means we have to cycle between the ready fds. 

802 # the first version used shuffle, but this version 

803 # using `total_writes % ready_fds` is about 30% faster 

804 # with many processes, and also leans more towards fairness 

805 # in write stats when used with many processes 

806 # [XXX On macOS, this may vary depending 

807 # on event loop implementation (i.e, select/poll vs epoll), so 

808 # have to test further] 

809 num_ready = len(ready_fds) 

810 

811 for _ in range(num_ready): 

812 ready_fd = ready_fds[total_write_count[0] % num_ready] 

813 total_write_count[0] += 1 

814 if ready_fd in active_writes: 

815 # already writing to this fd 

816 continue 

817 if is_fair_strategy and ready_fd in busy_workers: 

818 # worker is already busy with another task 

819 continue 

820 if ready_fd not in all_inqueues: 

821 hub_remove(ready_fd) 

822 continue 

823 try: 

824 job = pop_message() 

825 except IndexError: 

826 # no more messages, remove all inactive fds from the hub. 

827 # this is important since the fds are always writable 

828 # as long as there's 1 byte left in the buffer, and so 

829 # this may create a spinloop where the event loop 

830 # always wakes up. 

831 for inqfd in diff(active_writes): 

832 hub_remove(inqfd) 

833 break 

834 

835 else: 

836 if not job._accepted: # job not accepted by another worker 

837 try: 

838 # keep track of what process the write operation 

839 # was scheduled for. 

840 proc = job._scheduled_for = fileno_to_inq[ready_fd] 

841 except KeyError: 

842 # write was scheduled for this fd but the process 

843 # has since exited and the message must be sent to 

844 # another process. 

845 put_message(job) 

846 continue 

847 cor = _write_job(proc, ready_fd, job) 

848 job._writer = ref(cor) 

849 mark_write_gen_as_active(cor) 

850 mark_write_fd_as_active(ready_fd) 

851 mark_worker_as_busy(ready_fd) 

852 

853 # Try to write immediately, in case there's an error. 

854 try: 

855 next(cor) 

856 except StopIteration: 

857 pass 

858 except OSError as exc: 

859 if exc.errno != errno.EBADF: 

860 raise 

861 else: 

862 add_writer(ready_fd, cor) 

863 hub.consolidate_callback = schedule_writes

885 def _write_job(proc, fd, job): 

886 # writes job to the worker process. 

887 # Operation must complete if more than one byte of data 

888 # was written. If the broker connection is lost 

889 # and no data was written the operation shall be canceled. 

890 header, body, body_size = job._payload 

891 errors = 0 

892 try: 

893 # job result keeps track of what process the job is sent to. 

894 job._write_to = proc 

895 send = proc.send_job_offset 

896 

897 Hw = Bw = 0 

898 # write header 

899 while Hw < 4: 

900 try: 

901 Hw += send(header, Hw) 

902 except Exception as exc: # pylint: disable=broad-except 

903 if getattr(exc, 'errno', None) not in UNAVAIL: 

904 raise 

905 # suspend until more data 

906 errors += 1 

907 if errors > 100: 

908 on_not_recovering(proc, fd, job, exc) 

909 raise StopIteration() 

910 yield 

911 else: 

912 errors = 0 

913 

914 # write body 

915 while Bw < body_size: 

916 try: 

917 Bw += send(body, Bw) 

918 except Exception as exc: # pylint: disable=broad-except 

919 if getattr(exc, 'errno', None) not in UNAVAIL: 

920 raise 

921 # suspend until more data 

922 errors += 1 

923 if errors > 100: 

924 on_not_recovering(proc, fd, job, exc) 

925 raise StopIteration() 

926 yield 

927 else: 

928 errors = 0 

929 finally: 

930 hub_remove(fd) 

931 write_stats[proc.index] += 1 

932 # message written, so this fd is now available 

933 active_writes.discard(fd) 

934 write_generator_done(job._writer()) # is a weakref

987 def flush(self): 

988 if self._state == TERMINATE: 

989 return 

990 # cancel all tasks that haven't been accepted so that NACK is sent 

991 # if synack is enabled. 

992 if self.synack: 

993 for job in self._cache.values(): 

994 if not job._accepted: 

995 job._cancel() 

996 

997 # clear the outgoing buffer as the tasks will be redelivered by 

998 # the broker anyway. 

999 if self.outbound_buffer: 

1000 self.outbound_buffer.clear() 

1001 

1002 self.maintain_pool() 

1003 

1004 try: 

1005 # ...but we must continue writing the payloads we already started 

1006 # to keep message boundaries. 

1007 # The messages may be NACK'ed later if synack is enabled. 

1008 if self._state == RUN: 

1009 # flush outgoing buffers 

1010 intervals = fxrange(0.01, 0.1, 0.01, repeatlast=True) 

1011

1013 # This dict comprehension requires the walrus operator which is only available in 3.8.

1015 for job in self._cache.values(): 

1016 writer = _get_job_writer(job) 

1017 if writer is not None:

1019 

1020 if not self._active_writers:

1022 else: 

1023 while self._active_writers:

1027 gen_not_started(gen)):

1029 # discard the task, but we must also remove 

1030 # it from the Pool._cache. 

1031 try: 

1032 job = owned_by[gen] 

1033 except KeyError: 

1034 pass 

1035 else: 

1036 # removes from Pool._cache 

1037 job.discard() 

1038 self._active_writers.discard(gen) 

1039 else: 

1040 try: 

1041 job = owned_by[gen] 

1042 except KeyError: 

1043 pass 

1044 else: 

1045 job_proc = job._write_to 

1046 if job_proc._is_alive(): 

1047 self._flush_writer(job_proc, gen) 

1048 

1049 job.discard() 

1050 # workers may have exited in the meantime. 

1051 self.maintain_pool() 

1052 sleep(next(intervals)) # don't busyloop 

1053 finally: 

1054 self.outbound_buffer.clear() 

1055 self._active_writers.clear()
