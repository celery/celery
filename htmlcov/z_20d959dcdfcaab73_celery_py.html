141@click.pass_context 

142def celery(ctx, app, broker, result_backend, loader, config, workdir, 

143 no_color, quiet, version, skip_checks): 

144 """Celery command entrypoint.""" 

145 if version: 

146 click.echo(VERSION_BANNER) 

147 ctx.exit() 

148 elif ctx.invoked_subcommand is None: 

149 click.echo(ctx.get_help()) 

150 ctx.exit() 

151 

152 if loader: 

153 # Default app takes loader from this env (Issue #1066). 

154 os.environ['CELERY_LOADER'] = loader 

155 if broker: 

156 os.environ['CELERY_BROKER_URL'] = broker 

157 if result_backend: 

158 os.environ['CELERY_RESULT_BACKEND'] = result_backend 

159 if config: 

160 os.environ['CELERY_CONFIG_MODULE'] = config 

161 if skip_checks: 

162 os.environ['CELERY_SKIP_CHECKS'] = 'true' 

163 ctx.obj = CLIContext(app=app, no_color=no_color, workdir=workdir, 

164 quiet=quiet) 

165 

166 # User options 

167 worker.params.extend(ctx.obj.app.user_options.get('worker', [])) 

168 beat.params.extend(ctx.obj.app.user_options.get('beat', [])) 

169 events.params.extend(ctx.obj.app.user_options.get('events', [])) 

170 

171 for command in celery.commands.values(): 

172 command.params.extend(ctx.obj.app.user_options.get('preload', []))

209celery {info_name} {option_name} celeryapp <...>

211The support for this usage was removed in Celery 5.0. Instead you should use `{option_name}` as a global option:
