650 def autodiscover_tasks(self, packages=None, 

651 related_name='tasks', force=False): 

652 """Auto-discover task modules. 

653 

654 Searches a list of packages for a "tasks.py" module (or use 

655 related_name argument). 

656 

657 If the name is empty, this will be delegated to fix-ups (e.g., Django). 

658 

659 For example if you have a directory layout like this: 

660 

661 .. code-block:: text 

662 

663 foo/__init__.py 

664 tasks.py 

665 models.py 

666 

667 bar/__init__.py 

668 tasks.py 

669 models.py 

670 

671 baz/__init__.py 

672 models.py 

673 

674 Then calling ``app.autodiscover_tasks(['foo', 'bar', 'baz'])`` will 

675 result in the modules ``foo.tasks`` and ``bar.tasks`` being imported. 

676 

677 Arguments: 

678 packages (List[str]): List of packages to search. 

679 This argument may also be a callable, in which case the 

680 value returned is used (for lazy evaluation). 

681 related_name (Optional[str]): The name of the module to find. Defaults 

682 to "tasks": meaning "look for 'module.tasks' for every 

683 module in ``packages``.". If ``None`` will only try to import 

684 the package, i.e. "look for 'module'". 

685 force (bool): By default this call is lazy so that the actual 

686 auto-discovery won't happen until an application imports 

687 the default modules. Forcing will cause the auto-discovery 

688 to happen immediately. 

689 """ 

690 if force: 

691 return self._autodiscover_tasks(packages, related_name) 

692 signals.import_modules.connect(starpromise( 

693 self._autodiscover_tasks, packages, related_name, 

694 ), weak=False, sender=self)

696 def _autodiscover_tasks(self, packages, related_name, **kwargs): 

697 if packages: 

698 return self._autodiscover_tasks_from_names(packages, related_name) 

699 return self._autodiscover_tasks_from_fixups(related_name)

701 def _autodiscover_tasks_from_names(self, packages, related_name): 

702 # packages argument can be lazy 

703 return self.loader.autodiscover_tasks( 

704 packages() if callable(packages) else packages, related_name, 

705 )

707 def _autodiscover_tasks_from_fixups(self, related_name): 

708 return self._autodiscover_tasks_from_names([ 

709 pkg for fixup in self._fixups 

710 if hasattr(fixup, 'autodiscover_tasks') 

711 for pkg in fixup.autodiscover_tasks() 

712 ], related_name=related_name)

714 def send_task(self, name, args=None, kwargs=None, countdown=None,

721 root_id=None, parent_id=None, route_name=None, 

722 shadow=None, chain=None, task_type=None, replaced_task_nesting=0, **options): 

723 """Send task by name. 

724 

725 Supports the same arguments as :meth:`@-Task.apply_async`. 

726

761 "\"Setting the TTL to 0 causes messages to be expired upon " 

762 "reaching a queue unless they can be delivered to a "

764 "If this was unintended, please check the code which " 

765 "published this task."

773 if parent: 

774 if not root_id: 774 ↛ 776line 774 didn't jump to line 776, because the condition on line 774 was always true

775 root_id = parent.request.root_id or parent.request.id 

776 if not parent_id: 776 ↛ 779line 776 didn't jump to line 779, because the condition on line 776 was always true

779 if conf.task_inherit_parent_priority:

781 parent.request.delivery_info.get('priority')) 

782 

783 # alias for 'task_as_v2' 

784 message = amqp.create_task_message( 

785 task_id, name, args, kwargs, countdown, eta, group_id, group_index, 

786 expires, retries, chord, 

787 maybe_list(link), maybe_list(link_error), 

788 reply_to or self.thread_oid, time_limit, soft_time_limit, 

789 self.conf.task_send_sent_event, 

790 root_id, parent_id, shadow, chain, 

791 ignore_result=ignore_result, 

792 replaced_task_nesting=replaced_task_nesting, **options 

793 ) 

794 

795 stamped_headers = options.pop('stamped_headers', []) 

796 for stamp in stamped_headers: 

797 options.pop(stamp) 

798 

799 if connection: 

800 producer = amqp.Producer(connection, auto_declare=False) 

801 

802 with self.producer_or_acquire(producer) as P: 

803 with P.connection._reraise_as_library_errors(): 

804 if not ignore_result: 804 ↛ 806line 804 didn't jump to line 806, because the condition on line 804 was always true

810 # the old signature. 

811 result.ignored = ignore_result

815 parent, have_parent = self.current_worker_task, True 

816 if parent: 

817 parent.add_trail(result)

995 def _load_config(self): 

996 if isinstance(self.on_configure, Signal): 

997 self.on_configure.send(sender=self) 

998 else: 

999 # used to be a method pre 4.0 

1000 self.on_configure() 

1001 if self._config_source: 

1002 self.loader.config_from_object(self._config_source) 

1003 self.configured = True 

1004 settings = detect_settings( 

1005 self.prepare_config(self.loader.conf), self._preconf, 

1006 ignore_keys=self._preconf_set_by_auto, prefix=self.namespace, 

1007 ) 

1008 if self._conf is not None: 

1009 # replace in place, as someone may have referenced app.conf, 

1010 # done some changes, accessed a key, and then try to make more 

1011 # changes to the reference and not the finalized value. 

1012 self._conf.swap_with(settings) 

1013 else: 

1014 self._conf = settings 

1015 

1016 # load lazy config dict initializers. 

1017 pending_def = self._pending_defaults 

1018 while pending_def: 

1019 self._conf.add_defaults(maybe_evaluate(pending_def.popleft()())) 

1020 

1021 # load lazy periodic tasks 

1022 pending_beat = self._pending_periodic_tasks 

1023 while pending_beat: 

1024 periodic_task_args, periodic_task_kwargs = pending_beat.popleft() 

1025 self._add_periodic_task(*periodic_task_args, **periodic_task_kwargs)

1028 return self._conf

1043 def add_periodic_task(self, schedule, sig, 

1044 args=(), kwargs=(), name=None, **opts): 

1045 """ 

1046 Add a periodic task to beat schedule. 

1047 

1048 Celery beat store tasks based on `sig` or `name` if provided. Adding the 

1049 same signature twice make the second task override the first one. To 

1050 avoid the override, use distinct `name` for them. 

1051 """ 

1052 key, entry = self._sig_to_periodic_task_entry( 

1053 schedule, sig, args, kwargs, name, **opts) 

1054 if self.configured: 

1055 self._add_periodic_task(key, entry, name=name) 

1056 else: 

1057 self._pending_periodic_tasks.append([(key, entry), {"name": name}]) 

1058 return key

1060 def _sig_to_periodic_task_entry(self, schedule, sig, 

1061 args=(), kwargs=None, name=None, **opts): 

1062 kwargs = {} if not kwargs else kwargs 

1063 sig = (sig.clone(args, kwargs) 

1064 if isinstance(sig, abstract.CallableSignature) 

1065 else self.signature(sig.name, args, kwargs)) 

1066 return name or repr(sig), { 

1067 'schedule': schedule, 

1068 'task': sig.name, 

1069 'args': sig.args, 

1070 'kwargs': sig.kwargs, 

1071 'options': dict(sig.options, **opts), 

1072 }
