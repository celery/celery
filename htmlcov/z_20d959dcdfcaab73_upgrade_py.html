19def _slurp(filename): 

20 # TODO: Handle case when file does not exist 

21 with codecs.open(filename, 'r', 'utf-8') as read_fh: 

22 return [line for line in read_fh]

25def _compat_key(key, namespace='CELERY'): 

26 key = key.upper() 

27 if not key.startswith(namespace): 

28 key = '_'.join([namespace, key]) 

29 return key

32def _backup(filename, suffix='.orig'): 

33 lines = [] 

34 backup_filename = ''.join([filename, suffix]) 

35 print(f'writing backup to {backup_filename}...', 

36 file=sys.stderr) 

37 with codecs.open(filename, 'r', 'utf-8') as read_fh: 

38 with codecs.open(backup_filename, 'w', 'utf-8') as backup_fh: 

39 for line in read_fh: 

40 backup_fh.write(line) 

41 lines.append(line) 

42 return lines

45def _to_new_key(line, keyfilter=pass1, source=defaults._TO_NEW_KEY): 

46 # sort by length to avoid, for example, broker_transport overriding 

47 # broker_transport_options. 

48 for old_key in reversed(sorted(source, key=lambda x: len(x))): 

49 new_line = line.replace(old_key, keyfilter(source[old_key])) 

50 if line != new_line and 'CELERY_CELERY' not in new_line: 

51 return 1, new_line # only one match per line. 

52 return 0, line

73 """Migrate settings from Celery 3.x to Celery 4.x.""" 

74 lines = _slurp(filename) 

75 keyfilter = _compat_key if django or compat else pass1 

76 print(f'processing {filename}...', file=sys.stderr) 

77 # gives list of tuples: ``(did_change, line_contents)`` 

78 new_lines = [ 

79 _to_new_key(line, keyfilter) for line in lines 

80 ] 

81 if any(n[0] for n in new_lines): # did have changes 

82 if not no_backup: 

83 _backup(filename) 

84 with codecs.open(filename, 'w', 'utf-8') as write_fh: 

85 for _, line in new_lines: 

86 write_fh.write(line)
