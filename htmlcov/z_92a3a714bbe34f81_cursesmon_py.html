220 def readline(self, x, y): 

221 buffer = '' 

222 curses.echo() 

223 try: 

224 i = 0 

225 while 1: 

226 ch = self.win.getch(x, y + i) 

227 if ch != -1: 

228 if ch in (10, curses.KEY_ENTER): # enter 

229 break 

230 if ch in (27,): 

231 buffer = '' 

232 break 

233 buffer += chr(ch) 

234 i += 1 

235 finally: 

236 curses.noecho()

252 task = self.state.tasks[self.selected_task] 

253 info = task.info(extra=['state']) 

254 infoitems = [ 

255 ('args', info.pop('args', None)), 

256 ('kwargs', info.pop('kwargs', None)) 

257 ] + list(info.items())

259 if key is None:

261 value = str(value)

342 win = self.win

344 x = LEFT_BORDER_OFFSET 

345 y = blank_line = count(2)

348 win.bkgd(' ', curses.color_pair(1)) 

349 win.border() 

350 win.addstr(1, x, self.greet, curses.A_DIM | curses.color_pair(5)) 

351 next(blank_line) 

352 win.addstr(next(y), x, self.format_row('UUID', 'TASK', 

353 'WORKER', 'TIME', 'STATE'), 

354 curses.A_BOLD | curses.A_UNDERLINE) 

355 tasks = self.tasks 

356 if tasks: 

357 for row, (_, task) in enumerate(tasks): 

358 if row > self.display_height: 

359 break 

360 

361 if task.uuid: 

362 lineno = next(y) 

363 self.display_task_row(lineno, task) 

364 

365 # -- Footer 

366 next(blank_line) 

367 win.hline(my - 6, x, curses.ACS_HLINE, self.screen_width - 4) 

368 

369 # Selected Task Info 

370 if self.selected_task: 

371 win.addstr(my - 5, x, self.selected_str, curses.A_BOLD) 

372 info = 'Missing extended info' 

373 detail = '' 

374 try: 

375 selection = self.state.tasks[self.selected_task] 

376 except KeyError: 

377 pass 

378 else: 

379 info = selection.info() 

380 if 'runtime' in info:

512def evtop(app=None): # pragma: no cover 

513 """Start curses monitor.""" 

514 app = app_or_default(app) 

515 state = app.events.State() 

516 display = CursesMonitor(state, app) 

517 display.init_screen() 

518 refresher = DisplayThread(display) 

519 refresher.start() 

520 try: 

521 capture_events(app, state, display) 

522 except Exception: 

523 refresher.shutdown = True 

524 refresher.join() 

525 display.resetscreen() 

526 raise 

527 except (KeyboardInterrupt, SystemExit): 

528 refresher.shutdown = True 

529 refresher.join() 

530 display.resetscreen()
