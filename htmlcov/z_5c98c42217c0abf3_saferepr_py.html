1"""Streaming, truncating, non-recursive version of :func:`repr`. 

2 

3Differences from regular :func:`repr`: 

4 

5- Sets are represented the Python 3 way: ``{1, 2}`` vs ``set([1, 2])``. 

6- Unicode strings does not have the ``u'`` prefix, even on Python 2. 

7- Empty set formatted as ``set()`` (Python 3), not ``set([])`` (Python 2). 

8- Longs don't have the ``L`` suffix. 

9 

10Very slow with no limits, super quick with limits. 

11"""

24#: Node representing literal text. 

25#: - .value: is the literal text value 

26#: - .truncate: specifies if this text can be truncated, for things like 

27#: LIT_DICT_END this will be False, as we always display 

28#: the ending brackets, e.g: [[[1, 2, 3, ...,], ..., ]] 

29#: - .direction: If +1 the current level is increment by one, 

30#: if -1 the current level is decremented by one, and 

31#: if 0 the current level is unchanged. 

32_literal = namedtuple('_literal', ('value', 'truncate', 'direction')) 

33 

34#: Node representing a dictionary key. 

35_key = namedtuple('_key', ('value',)) 

36 

37#: Node representing quoted text, e.g. a string value. 

38_quoted = namedtuple('_quoted', ('value',)) 

39 

40 

41#: Recursion protection. 

42_dirty = namedtuple('_dirty', ('objid',)) 

43 

44#: Types that are repsented as chars. 

45chars_t = (bytes, str) 

46 

47#: Types that are regarded as safe to call repr on. 

48safe_t = (Number,) 

49 

50#: Set types. 

51set_t = (frozenset, set)

66def saferepr(o, maxlen=None, maxlevels=3, seen=None): 

67 # type: (Any, int, int, Set) -> str 

68 """Safe version of :func:`repr`. 

69 

70 Warning: 

71 Make sure you set the maxlen argument, or it will be very slow 

72 for recursive objects. With the maxlen set, it's often faster 

73 than built-in repr. 

74 """ 

75 return ''.join(_saferepr( 

76 o, maxlen=maxlen, maxlevels=maxlevels, seen=seen 

77 ))

80def _chaindict(mapping, 

81 LIT_DICT_KVSEP=LIT_DICT_KVSEP, 

82 LIT_LIST_SEP=LIT_LIST_SEP): 

83 # type: (Dict, _literal, _literal) -> Iterator[Any] 

84 size = len(mapping) 

85 for i, (k, v) in enumerate(mapping.items()): 

86 yield _key(k) 

87 yield LIT_DICT_KVSEP 

88 yield v 

89 if i < (size - 1): 

90 yield LIT_LIST_SEP

93def _chainlist(it, LIT_LIST_SEP=LIT_LIST_SEP): 

94 # type: (List) -> Iterator[Any] 

95 size = len(it) 

96 for i, v in enumerate(it): 

97 yield v 

98 if i < (size - 1): 

99 yield LIT_LIST_SEP

110 try:

119def _format_binary_bytes(val, maxlen, ellipsis='...'): 

120 # type: (bytes, int, str) -> str 

121 if maxlen and len(val) > maxlen: 

122 # we don't want to copy all the data, just take what we need. 

123 chunk = memoryview(val)[:maxlen].tobytes() 

124 return _bytes_prefix(f"'{_repr_binary_bytes(chunk)}{ellipsis}'") 

125 return _bytes_prefix(f"'{_repr_binary_bytes(val)}'")

141def _format_chars(val, maxlen): 

142 # type: (AnyStr, int) -> str 

143 if isinstance(val, bytes): # pragma: no cover 

144 return _format_binary_bytes(val, maxlen) 

145 else: 

146 return "'{}'".format(truncate(val, maxlen).replace("'", "\\'"))

149def _repr(obj): 

150 # type: (Any) -> str 

151 try: 

152 return repr(obj) 

153 except Exception as exc: 

154 stack = '\n'.join(traceback.format_stack()) 

155 return f'<Unrepresentable {type(obj)!r}{id(obj):#x}: {exc!r} {stack!r}>'

158def _saferepr(o, maxlen=None, maxlevels=3, seen=None): 

159 # type: (Any, int, int, Set) -> str 

160 stack = deque([iter([o])]) 

161 for token, it in reprstream(stack, seen=seen, maxlevels=maxlevels): 

162 if maxlen is not None and maxlen <= 0: 

163 yield ', ...' 

164 # move rest back to stack, so that we can include 

165 # dangling parens. 

166 stack.append(it) 

167 break 

168 if isinstance(token, _literal): 

169 val = token.value 

170 elif isinstance(token, _key): 

171 val = saferepr(token.value, maxlen, maxlevels) 

172 elif isinstance(token, _quoted): 

173 val = _format_chars(token.value, maxlen) 

174 else: 

175 val = _safetext(truncate(token, maxlen)) 

176 yield val 

177 if maxlen is not None: 

178 maxlen -= len(val) 

179 for rest1 in stack: 

180 # maxlen exceeded, process any dangling parens. 

181 for rest2 in rest1: 

182 if isinstance(rest2, _literal) and not rest2.truncate: 

183 yield rest2.value

186def _reprseq(val, lit_start, lit_end, builtin_type, chainer): 

187 # type: (Sequence, _literal, _literal, Any, Any) -> Tuple[Any, ...] 

188 if type(val) is builtin_type: 

189 return lit_start, lit_end, chainer(val) 

190 return ( 

191 _literal(f'{type(val).__name__}({lit_start.value}', False, +1), 

192 _literal(f'{lit_end.value})', False, -1), 

193 chainer(val) 

194 )

197def reprstream(stack: deque, 

198 seen: Optional[Set] = None, 

199 maxlevels: int = 3, 

200 level: int = 0, 

201 isinstance: Callable = isinstance) -> Iterator[Any]: 

202 """Streaming repr, yielding tokens.""" 

203 seen = seen or set() 

204 append = stack.append 

205 popleft = stack.popleft 

206 is_in_seen = seen.__contains__ 

207 discard_from_seen = seen.discard 

208 add_to_seen = seen.add 

209 

210 while stack: 

211 lit_start = lit_end = None 

212 it = popleft() 

213 for val in it: 

214 orig = val 

215 if isinstance(val, _dirty): 

216 discard_from_seen(val.objid) 

217 continue 

218 elif isinstance(val, _literal): 

219 level += val.direction 

220 yield val, it 

221 elif isinstance(val, _key): 

222 yield val, it 

223 elif isinstance(val, Decimal): 

224 yield _repr(val), it 

225 elif isinstance(val, safe_t): 

226 yield str(val), it 

227 elif isinstance(val, chars_t): 

228 yield _quoted(val), it 

229 elif isinstance(val, range): # pragma: no cover 

230 yield _repr(val), it 

231 else: 

232 if isinstance(val, set_t): 

233 if not val: 

234 yield _repr_empty_set(val), it 

235 continue 

236 lit_start, lit_end, val = _reprseq( 

237 val, LIT_SET_START, LIT_SET_END, set, _chainlist, 

238 ) 

239 elif isinstance(val, tuple): 

240 lit_start, lit_end, val = ( 

241 LIT_TUPLE_START, 

242 LIT_TUPLE_END_SV if len(val) == 1 else LIT_TUPLE_END, 

243 _chainlist(val)) 

244 elif isinstance(val, dict): 

245 lit_start, lit_end, val = ( 

246 LIT_DICT_START, LIT_DICT_END, _chaindict(val)) 

247 elif isinstance(val, list): 

248 lit_start, lit_end, val = ( 

249 LIT_LIST_START, LIT_LIST_END, _chainlist(val)) 

250 else: 

251 # other type of object 

252 yield _repr(val), it 

253 continue 

254 

255 if maxlevels and level >= maxlevels: 

256 yield f'{lit_start.value}...{lit_end.value}', it 

257 continue 

258 

259 objid = id(orig) 

260 if is_in_seen(objid): 

261 yield _recursion(orig), it 

262 continue 

263 add_to_seen(objid) 

264 

265 # Recurse into the new list/tuple/dict/etc by tacking 

266 # the rest of our iterable onto the new it: this way 

267 # it works similar to a linked list. 

268 append(chain([lit_start], val, [_dirty(objid), lit_end], it)) 

269 break
