651 Example: 

652 >>> from imaginary_twitter_lib import Twitter 

653 >>> from proj.celery import app 

654 

655 >>> @app.task(bind=True) 

656 ... def tweet(self, auth, message): 

657 ... twitter = Twitter(oauth=auth) 

658 ... try: 

659 ... twitter.post_status_update(message) 

660 ... except twitter.FailWhale as exc: 

661 ... # Retry in 5 minutes. 

662 ... raise self.retry(countdown=60 * 5, exc=exc) 

663 

664 Note: 

665 Although the task will never return above as `retry` raises an 

666 exception to notify the worker, we use `raise` in front of the 

667 retry to convey that the rest of the block won't be executed. 

668 

669 Arguments: 

670 args (Tuple): Positional arguments to retry with. 

671 kwargs (Dict): Keyword arguments to retry with. 

672 exc (Exception): Custom exception to report when the max retry 

673 limit has been exceeded (default: 

674 :exc:`~@MaxRetriesExceededError`). 

675 

676 If this argument is set and retry is called while 

677 an exception was raised (``sys.exc_info()`` is set) 

678 it will attempt to re-raise the current exception. 

679 

680 If no exception was raised it will raise the ``exc`` 

681 argument provided. 

682 countdown (float): Time in seconds to delay the retry for. 

683 eta (~datetime.datetime): Explicit time and date to run the 

684 retry at. 

685 max_retries (int): If set, overrides the default retry limit for 

686 this execution. Changes to this parameter don't propagate to 

687 subsequent task retry attempts. A value of :const:`None`, 

688 means "use the default", so if you want infinite retries you'd 

689 have to set the :attr:`max_retries` attribute of the task to 

690 :const:`None` first. 

691 time_limit (int): If set, overrides the default time limit. 

692 soft_time_limit (int): If set, overrides the default soft 

693 time limit. 

694 throw (bool): If this is :const:`False`, don't raise the 

695 :exc:`~@Retry` exception, that tells the worker to mark 

696 the task as being retried. Note that this means the task 

697 will be marked as failed if the task raises an exception, 

698 or successful if it returns after the retry call. 

699 **options (Any): Extra options to pass on to :meth:`apply_async`. 

700 

701 Raises: 

702 

703 celery.exceptions.Retry: 

704 To tell the worker that the task has been re-sent for retry. 

705 This always happens, unless the `throw` keyword argument 

706 has been explicitly set to :const:`False`, and is considered 

707 normal operation. 

708 """ 

709 request = self.request 

710 retries = request.retries + 1 

711 if max_retries is not None: 

712 self.override_max_retries = max_retries 

713 max_retries = self.max_retries if max_retries is None else max_retries 

714 

715 # Not in worker or emulated by (apply/always_eager), 

716 # so just raise the original exception. 

717 if request.called_directly: 

718 # raises orig stack if PyErr_Occurred, 

719 # and augments with exc' if that argument is defined. 

720 raise_with_context(exc or Retry('Task can be retried', None)) 

721 

722 if not eta and countdown is None: 

723 countdown = self.default_retry_delay 

724 

725 is_eager = request.is_eager 

726 S = self.signature_from_request( 

727 request, args, kwargs, 

728 countdown=countdown, eta=eta, retries=retries, 

729 **options 

730 ) 

731 

732 if max_retries is not None and retries > max_retries: 

733 if exc: 

734 # On Py3: will augment any current exception with 

735 # the exc' argument provided (raise exc from orig) 

736 raise_with_context(exc) 

737 raise self.MaxRetriesExceededError( 

738 "Can't retry {}[{}] args:{} kwargs:{}".format( 

739 self.name, request.id, S.args, S.kwargs 

740 ), task_args=S.args, task_kwargs=S.kwargs 

741 ) 

742 

743 ret = Retry(exc=exc, when=eta or countdown, is_eager=is_eager, sig=S) 

744 

745 if is_eager: 

746 # if task was executed eagerly using apply(), 

747 # then the retry must also be executed eagerly in apply method 

748 if throw: 

749 raise ret 

750 return ret 

751 

752 try: 

753 S.apply_async() 

754 except Exception as exc: 

755 raise Reject(exc, requeue=False) 

756 if throw: 

757 raise ret 

758 return ret

760 def apply(self, args=None, kwargs=None, 

761 link=None, link_error=None, 

762 task_id=None, retries=None, throw=None, 

763 logfile=None, loglevel=None, headers=None, **options): 

764 """Execute this task locally, by blocking until the task returns. 

765 

766 Arguments: 

767 args (Tuple): positional arguments passed on to the task. 

768 kwargs (Dict): keyword arguments passed on to the task. 

769 throw (bool): Re-raise task exceptions. 

770 Defaults to the :setting:`task_eager_propagates` setting. 

771 

772 Returns: 

773 celery.result.EagerResult: pre-evaluated result. 

774 """ 

775 # trace imports Task, so need to import inline. 

776 from celery.app.trace import build_tracer 

777 

778 app = self._get_app() 

779 args = args or () 

780 kwargs = kwargs or {} 

781 task_id = task_id or uuid() 

782 retries = retries or 0 

783 if throw is None: 

784 throw = app.conf.task_eager_propagates 

785 

786 # Make sure we get the task instance, not class. 

787 task = app._tasks[self.name] 

788 

789 request = { 

790 'id': task_id, 

791 'task': self.name, 

792 'retries': retries, 

793 'is_eager': True, 

794 'logfile': logfile, 

795 'loglevel': loglevel or 0, 

796 'hostname': gethostname(), 

797 'callbacks': maybe_list(link), 

798 'errbacks': maybe_list(link_error), 

799 'headers': headers, 

800 'ignore_result': options.get('ignore_result', False), 

801 'delivery_info': { 

802 'is_eager': True, 

803 'exchange': options.get('exchange'), 

804 'routing_key': options.get('routing_key'), 

805 'priority': options.get('priority'), 

806 } 

807 } 

808 if 'stamped_headers' in options: 

809 request['stamped_headers'] = maybe_list(options['stamped_headers']) 

810 request['stamps'] = { 

811 header: maybe_list(options.get(header, [])) for header in request['stamped_headers'] 

812 } 

813 

814 tb = None 

815 tracer = build_tracer( 

816 task.name, task, eager=True, 

817 propagate=throw, app=self._get_app(), 

818 ) 

819 ret = tracer(task_id, args, kwargs, request) 

820 retval = ret.retval 

821 if isinstance(retval, ExceptionInfo): 

822 retval, tb = retval.exception, retval.traceback 

823 if isinstance(retval, ExceptionWithTraceback): 

824 retval = retval.exc 

825 if isinstance(retval, Retry) and retval.sig is not None: 

826 return retval.sig.apply(retries=retries + 1) 

827 state = states.SUCCESS if ret.info is None else ret.info.state 

828 return EagerResult(task_id, retval, state, traceback=tb, name=self.name)

906 def replace(self, sig): 

907 """Replace this task, with a new task inheriting the task id. 

908 

909 Execution of the host task ends immediately and no subsequent statements 

910 will be run. 

911 

912 .. versionadded:: 4.0 

913 

914 Arguments: 

915 sig (Signature): signature to replace with. 

916 visitor (StampingVisitor): Visitor API object. 

917 

918 Raises: 

919 ~@Ignore: This is always raised when called in asynchronous context. 

920 It is best to always use ``return self.replace(...)`` to convey 

921 to the reader that the task won't continue after being replaced. 

922 """ 

923 chord = self.request.chord 

924 if 'chord' in sig.options: 

925 raise ImproperlyConfigured( 

926 "A signature replacing a task must not be part of a chord" 

927 ) 

928 if isinstance(sig, _chain) and not getattr(sig, "tasks", True): 928 ↛ 929line 928 didn't jump to line 929, because the condition on line 928 was never true

929 raise ImproperlyConfigured("Cannot replace with an empty chain") 

930 

931 # Ensure callbacks or errbacks from the replaced signature are retained 

932 if isinstance(sig, group): 

933 # Groups get uplifted to a chord so that we can link onto the body 

934 sig |= self.app.tasks['celery.accumulate'].s(index=0) 

935 for callback in maybe_list(self.request.callbacks) or []: 

936 sig.link(callback) 

937 for errback in maybe_list(self.request.errbacks) or []: 

938 sig.link_error(errback) 

939 # If the replacement signature is a chain, we need to push callbacks 

940 # down to the final task so they run at the right time even if we 

941 # proceed to link further tasks from the original request below 

942 if isinstance(sig, _chain) and "link" in sig.options: 942 ↛ 943line 942 didn't jump to line 943, because the condition on line 942 was never true

943 final_task_links = sig.tasks[-1].options.setdefault("link", []) 

944 final_task_links.extend(maybe_list(sig.options["link"])) 

945 # We need to freeze the replacement signature with the current task's 

946 # ID to ensure that we don't disassociate it from the existing task IDs 

947 # which would break previously constructed results objects. 

948 sig.freeze(self.request.id) 

949 # Ensure the important options from the original signature are retained 

950 replaced_task_nesting = self.request.get('replaced_task_nesting', 0) + 1 

951 sig.set( 

952 chord=chord, 

953 group_id=self.request.group, 

954 group_index=self.request.group_index, 

955 root_id=self.request.root_id, 

956 replaced_task_nesting=replaced_task_nesting 

957 ) 

958 

959 # If the replaced task is a chain, we want to set all of the chain tasks 

960 # with the same replaced_task_nesting value to mark their replacement nesting level 

961 if isinstance(sig, _chain): 

962 for chain_task in maybe_list(sig.tasks) or []: 

963 chain_task.set(replaced_task_nesting=replaced_task_nesting) 

964 

965 # If the task being replaced is part of a chain, we need to re-create 

966 # it with the replacement signature - these subsequent tasks will 

967 # retain their original task IDs as well 

968 for t in reversed(self.request.chain or []): 

969 chain_task = signature(t, app=self.app) 

970 chain_task.set(replaced_task_nesting=replaced_task_nesting) 

971 sig |= chain_task 

972 return self.on_replace(sig)
