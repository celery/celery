32def _boundmethod_safe_weakref(obj): 

33 """Get weakref constructor appropriate for `obj`. `obj` may be a bound method. 

34 

35 Bound method objects must be special-cased because they're usually garbage 

36 collected immediately, even if the instance they're bound to persists. 

37 

38 Returns: 

39 a (weakref constructor, main object) tuple. `weakref constructor` is 

40 either :class:`weakref.ref` or :class:`weakref.WeakMethod`. `main 

41 object` is the instance that `obj` is bound to if it is a bound method; 

42 otherwise `main object` is simply `obj. 

43 """ 

44 try: 

45 obj.__func__ 

46 obj.__self__ 

47 # Bound method 

48 return WeakMethod, obj.__self__ 

49 except AttributeError: 

50 # Not a bound method 

51 return weakref.ref, obj

107 def connect(self, *args, **kwargs): 

108 """Connect receiver to sender for signal. 

109 

110 Arguments: 

111 receiver (Callable): A function or an instance method which is to 

112 receive signals. Receivers must be hashable objects. 

113 

114 if weak is :const:`True`, then receiver must be 

115 weak-referenceable. 

116 

117 Receivers must be able to accept keyword arguments. 

118 

119 If receivers have a `dispatch_uid` attribute, the receiver will 

120 not be added if another receiver already exists with that 

121 `dispatch_uid`. 

122 

123 sender (Any): The sender to which the receiver should respond. 

124 Must either be a Python object, or :const:`None` to 

125 receive events from any sender. 

126 

127 weak (bool): Whether to use weak references to the receiver. 

128 By default, the module will attempt to use weak references to 

129 the receiver objects. If this parameter is false, then strong 

130 references will be used. 

131 

132 dispatch_uid (Hashable): An identifier used to uniquely identify a 

133 particular instance of a receiver. This will usually be a 

134 string, though it may be anything hashable. 

135 

136 retry (bool): If the signal receiver raises an exception 

137 (e.g. ConnectionError), the receiver will be retried until it 

138 runs successfully. A strong ref to the receiver will be stored 

139 and the `weak` option will be ignored. 

140 """

184 def _connect_signal(self, receiver, sender, weak, dispatch_uid): 

185 assert callable(receiver), 'Signal receivers must be callable' 

186 if not fun_accepts_kwargs(receiver): 

187 raise ValueError( 

188 'Signal receiver must accept keyword arguments.') 

189 

190 if isinstance(sender, PromiseProxy): 

191 sender.__then__( 

192 self._connect_proxy, receiver, sender, weak, dispatch_uid, 

193 ) 

194 return receiver 

195 

196 lookup_key = _make_lookup_key(receiver, sender, dispatch_uid) 

197 

198 if weak: 

199 ref, receiver_object = _boundmethod_safe_weakref(receiver) 

200 receiver = ref(receiver) 

201 weakref.finalize(receiver_object, self._remove_receiver) 

202 

203 with self.lock: 

204 self._clear_dead_receivers() 

205 for r_key, _ in self.receivers: 

206 if r_key == lookup_key: 

207 break 

208 else: 

209 self.receivers.append((lookup_key, receiver)) 

210 self.sender_receivers_cache.clear() 

211 

212 return receiver

214 def disconnect(self, receiver=None, sender=None, weak=None, 

215 dispatch_uid=None): 

216 """Disconnect receiver from sender for signal. 

217 

218 If weak references are used, disconnect needn't be called. 

219 The receiver will be removed from dispatch automatically. 

220 

221 Arguments: 

222 receiver (Callable): The registered receiver to disconnect. 

223 May be none if `dispatch_uid` is specified. 

224 

225 sender (Any): The registered sender to disconnect. 

226 

227 weak (bool): The weakref state to disconnect. 

228 

229 dispatch_uid (Hashable): The unique identifier of the receiver 

230 to disconnect. 

231 """ 

232 if weak is not None: 

233 warnings.warn( 

234 'Passing `weak` to disconnect has no effect.', 

235 CDeprecationWarning, stacklevel=2) 

236 

237 lookup_key = _make_lookup_key(receiver, sender, dispatch_uid) 

238 

239 disconnected = False 

240 with self.lock: 

241 self._clear_dead_receivers() 

242 for index in range(len(self.receivers)): 

243 (r_key, _) = self.receivers[index] 

244 if r_key == lookup_key: 

245 disconnected = True 

246 del self.receivers[index] 

247 break 

248 self.sender_receivers_cache.clear() 

249 return disconnected

254 def send(self, sender, **named): 

255 """Send signal from sender to all connected receivers. 

256 

257 If any receiver raises an error, the exception is returned as the 

258 corresponding response. (This is different from the "send" in 

259 Django signals. In Celery "send" and "send_robust" do the same thing.) 

260 

261 Arguments: 

262 sender (Any): The sender of the signal. 

263 Either a specific object or :const:`None`. 

264 **named (Any): Named arguments which will be passed to receivers. 

265 

266 Returns: 

267 List: of tuple pairs: `[(receiver, response), … ]`. 

268 """ 

269 responses = [] 

270 if not self.receivers or \ 

271 self.sender_receivers_cache.get(sender) is NO_RECEIVERS: 

272 return responses 

273 

274 for receiver in self._live_receivers(sender): 

275 try: 

276 response = receiver(signal=self, sender=sender, **named) 

277 except Exception as exc: # pylint: disable=broad-except 

278 if not hasattr(exc, '__traceback__'): 

279 exc.__traceback__ = sys.exc_info()[2] 

280 logger.exception( 

281 'Signal handler %r raised: %r', receiver, exc) 

282 responses.append((receiver, exc)) 

283 else: 

284 responses.append((receiver, response)) 

285 return responses

288 def _clear_dead_receivers(self):

299 def _live_receivers(self, sender): 

300 """Filter sequence of receivers to get resolved, live receivers. 

301 

302 This checks for weak references and resolves them, then returning only 

303 live receivers. 

304 """ 

305 receivers = None 

306 if self.use_caching and not self._dead_receivers: 

307 receivers = self.sender_receivers_cache.get(sender) 

308 # We could end up here with NO_RECEIVERS even if we do check this 

309 # case in .send() prior to calling _Live_receivers() due to 

310 # concurrent .send() call. 

311 if receivers is NO_RECEIVERS: 

312 return [] 

313 if receivers is None: 

314 with self.lock: 

315 self._clear_dead_receivers() 

316 senderkey = _make_id(sender) 

317 receivers = [] 

318 for (receiverkey, r_senderkey), receiver in self.receivers: 

319 if r_senderkey == NONE_ID or r_senderkey == senderkey: 

320 receivers.append(receiver) 

321 if self.use_caching: 

322 if not receivers: 

323 self.sender_receivers_cache[sender] = NO_RECEIVERS 

324 else: 

325 # Note: we must cache the weakref versions. 

326 self.sender_receivers_cache[sender] = receivers 

327 non_weak_receivers = [] 

328 for receiver in receivers: 

329 if isinstance(receiver, weakref.ReferenceType): 

330 # Dereference the weak reference. 

331 receiver = receiver() 

332 if receiver is not None: 

333 non_weak_receivers.append(receiver) 

334 else: 

335 non_weak_receivers.append(receiver) 

336 return non_weak_receivers

338 def _remove_receiver(self, receiver=None): 

339 """Remove dead receivers from connections.""" 

340 # Mark that the self..receivers first has dead weakrefs. If so, 

341 # we will clean those up in connect, disconnect and _live_receivers 

342 # while holding self.lock. Note that doing the cleanup here isn't a 

343 # good idea, _remove_receiver() will be called as a side effect of 

344 # garbage collection, and so the call can happen wh ile we are already 

345 # holding self.lock. 

346 self._dead_receivers = True
