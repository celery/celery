136 def is_due(self, last_run_at: datetime) -> tuple[bool, datetime]: 

137 """Return tuple of ``(is_due, next_time_to_check)``. 

138 

139 Notes: 

140 - next time to check is in seconds. 

141 

142 - ``(True, 20)``, means the task should be run now, and the next 

143 time to check is in 20 seconds. 

144 

145 - ``(False, 12.3)``, means the task is not due, but that the 

146 scheduler should check again in 12.3 seconds. 

147 

148 The next time to check is used to save energy/CPU cycles, 

149 it does not need to be accurate but will influence the precision 

150 of your schedule. You must also keep in mind 

151 the value of :setting:`beat_max_loop_interval`, 

152 that decides the maximum number of seconds the scheduler can 

153 sleep between re-checking the periodic task intervals. So if you 

154 have a task that changes schedule at run-time then your next_run_at 

155 check will decide how long it will take before a change to the 

156 schedule takes effect. The max loop interval takes precedence 

157 over the next check at value returned. 

158 

159 .. admonition:: Scheduler max interval variance 

160 

161 The default max loop interval may vary for different schedulers. 

162 For the default scheduler the value is 5 minutes, but for example 

163 the :pypi:`django-celery-beat` database scheduler the value 

164 is 5 seconds. 

165 """ 

166 last_run_at = self.maybe_make_aware(last_run_at) 

167 rem_delta = self.remaining_estimate(last_run_at) 

168 remaining_s = max(rem_delta.total_seconds(), 0) 

169 if remaining_s == 0: 

170 return schedstate(is_due=True, next=self.seconds) 

171 return schedstate(is_due=False, next=remaining_s)

194class crontab_parser: 

195 """Parser for Crontab expressions. 

196 

197 Any expression of the form 'groups' 

198 (see BNF grammar below) is accepted and expanded to a set of numbers. 

199 These numbers represent the units of time that the Crontab needs to 

200 run on: 

201 

202 .. code-block:: bnf 

203 

204 digit :: '0'..'9' 

205 dow :: 'a'..'z' 

206 number :: digit+ | dow+ 

207 steps :: number 

208 range :: number ( '-' number ) ? 

209 numspec :: '*' | range 

210 expr :: numspec ( '/' steps ) ? 

211 groups :: expr ( ',' expr ) * 

212 

213 The parser is a general purpose one, useful for parsing hours, minutes and 

214 day of week expressions. Example usage: 

215 

216 .. code-block:: pycon 

217 

218 >>> minutes = crontab_parser(60).parse('*/15') 

219 [0, 15, 30, 45] 

220 >>> hours = crontab_parser(24).parse('*/4') 

221 [0, 4, 8, 12, 16, 20] 

222 >>> day_of_week = crontab_parser(7).parse('*') 

223 [0, 1, 2, 3, 4, 5, 6] 

224 

225 It can also parse day of month and month of year expressions if initialized 

226 with a minimum of 1. Example usage: 

227 

228 .. code-block:: pycon 

229 

230 >>> days_of_month = crontab_parser(31, 1).parse('*/3') 

231 [1, 4, 7, 10, 13, 16, 19, 22, 25, 28, 31] 

232 >>> months_of_year = crontab_parser(12, 1).parse('*/2') 

233 [1, 3, 5, 7, 9, 11] 

234 >>> months_of_year = crontab_parser(12, 1).parse('2-12/2') 

235 [2, 4, 6, 8, 10, 12] 

236 

237 The maximum possible expanded value returned is found by the formula: 

238 

239 :math:`max_ + min_ - 1` 

240 """ 

241 

242 ParseException = ParseException 

243 

244 _range = r'(\w+?)-(\w+)' 

245 _steps = r'/(\w+)?' 

246 _star = r'\*' 

247 

248 def __init__(self, max_: int = 60, min_: int = 0): 

249 self.max_ = max_ 

250 self.min_ = min_ 

251 self.pats: tuple[tuple[re.Pattern, Callable], ...] = ( 

252 (re.compile(self._range + self._steps), self._range_steps), 

253 (re.compile(self._range), self._expand_range), 

254 (re.compile(self._star + self._steps), self._star_steps), 

255 (re.compile('^' + self._star + '$'), self._expand_star), 

256 ) 

257 

258 def parse(self, spec: str) -> set[int]: 

259 acc = set() 

260 for part in spec.split(','): 

261 if not part: 

262 raise self.ParseException('empty part') 

263 acc |= set(self._parse_part(part)) 

264 return acc 

265 

266 def _parse_part(self, part: str) -> list[int]: 

267 for regex, handler in self.pats: 

268 m = regex.match(part) 

269 if m: 

270 return handler(m.groups()) 

271 return self._expand_range((part,)) 

272 

273 def _expand_range(self, toks: Sequence[str]) -> list[int]: 

274 fr = self._expand_number(toks[0]) 

275 if len(toks) > 1: 

276 to = self._expand_number(toks[1]) 

277 if to < fr: # Wrap around max_ if necessary 

278 return (list(range(fr, self.min_ + self.max_)) + 

279 list(range(self.min_, to + 1))) 

280 return list(range(fr, to + 1)) 

281 return [fr] 

282 

283 def _range_steps(self, toks: Sequence[str]) -> list[int]: 

284 if len(toks) != 3 or not toks[2]: 

285 raise self.ParseException('empty filter') 

286 return self._expand_range(toks[:2])[::int(toks[2])] 

287 

288 def _star_steps(self, toks: Sequence[str]) -> list[int]: 

289 if not toks or not toks[0]: 

290 raise self.ParseException('empty filter') 

291 return self._expand_star()[::int(toks[0])] 

292 

293 def _expand_star(self, *args: Any) -> list[int]: 

294 return list(range(self.min_, self.max_ + self.min_)) 

295 

296 def _expand_number(self, s: str) -> int: 

297 if isinstance(s, str) and s[0] == '-': 

298 raise self.ParseException('negative numbers not supported') 

299 try: 

300 i = int(s) 

301 except ValueError: 

302 try: 

303 i = yearmonth(s) 

304 except KeyError: 

305 try: 

306 i = weekday(s) 

307 except KeyError: 

308 raise ValueError(f'Invalid weekday literal {s!r}.') 

309 

310 max_val = self.min_ + self.max_ - 1 

311 if i > max_val: 

312 raise ValueError( 

313 f'Invalid end range: {i} > {max_val}.') 

314 if i < self.min_: 

315 raise ValueError( 

316 f'Invalid beginning range: {i} < {self.min_}.') 

317 

318 return i

321class crontab(BaseSchedule): 

322 """Crontab schedule. 

323 

324 A Crontab can be used as the ``run_every`` value of a 

325 periodic task entry to add :manpage:`crontab(5)`-like scheduling. 

326 

327 Like a :manpage:`cron(5)`-job, you can specify units of time of when 

328 you'd like the task to execute. It's a reasonably complete 

329 implementation of :command:`cron`'s features, so it should provide a fair 

330 degree of scheduling needs. 

331 

332 You can specify a minute, an hour, a day of the week, a day of the 

333 month, and/or a month in the year in any of the following formats: 

334 

335 .. attribute:: minute 

336 

337 - A (list of) integers from 0-59 that represent the minutes of 

338 an hour of when execution should occur; or 

339 - A string representing a Crontab pattern. This may get pretty 

340 advanced, like ``minute='*/15'`` (for every quarter) or 

341 ``minute='1,13,30-45,50-59/2'``. 

342 

343 .. attribute:: hour 

344 

345 - A (list of) integers from 0-23 that represent the hours of 

346 a day of when execution should occur; or 

347 - A string representing a Crontab pattern. This may get pretty 

348 advanced, like ``hour='*/3'`` (for every three hours) or 

349 ``hour='0,8-17/2'`` (at midnight, and every two hours during 

350 office hours). 

351 

352 .. attribute:: day_of_week 

353 

354 - A (list of) integers from 0-6, where Sunday = 0 and Saturday = 

355 6, that represent the days of a week that execution should 

356 occur. 

357 - A string representing a Crontab pattern. This may get pretty 

358 advanced, like ``day_of_week='mon-fri'`` (for weekdays only). 

359 (Beware that ``day_of_week='*/2'`` does not literally mean 

360 'every two days', but 'every day that is divisible by two'!) 

361 

362 .. attribute:: day_of_month 

363 

364 - A (list of) integers from 1-31 that represents the days of the 

365 month that execution should occur. 

366 - A string representing a Crontab pattern. This may get pretty 

367 advanced, such as ``day_of_month='2-30/2'`` (for every even 

368 numbered day) or ``day_of_month='1-7,15-21'`` (for the first and 

369 third weeks of the month). 

370 

371 .. attribute:: month_of_year 

372 

373 - A (list of) integers from 1-12 that represents the months of 

374 the year during which execution can occur. 

375 - A string representing a Crontab pattern. This may get pretty 

376 advanced, such as ``month_of_year='*/3'`` (for the first month 

377 of every quarter) or ``month_of_year='2-12/2'`` (for every even 

378 numbered month). 

379 

380 .. attribute:: nowfun 

381 

382 Function returning the current date and time 

383 (:class:`~datetime.datetime`). 

384 

385 .. attribute:: app 

386 

387 The Celery app instance. 

388 

389 It's important to realize that any day on which execution should 

390 occur must be represented by entries in all three of the day and 

391 month attributes. For example, if ``day_of_week`` is 0 and 

392 ``day_of_month`` is every seventh day, only months that begin 

393 on Sunday and are also in the ``month_of_year`` attribute will have 

394 execution events. Or, ``day_of_week`` is 1 and ``day_of_month`` 

395 is '1-7,15-21' means every first and third Monday of every month 

396 present in ``month_of_year``. 

397 """

415 def _expand_cronspec( 

416 cronspec: int | str | Iterable, 

417 max_: int, min_: int = 0) -> set[Any]: 

418 """Expand cron specification. 

419 

420 Takes the given cronspec argument in one of the forms: 

421 

422 .. code-block:: text 

423 

424 int (like 7) 

425 str (like '3-5,*/15', '*', or 'monday') 

426 set (like {0,15,30,45} 

427 list (like [8-17]) 

428 

429 And convert it to an (expanded) set representing all time unit 

430 values on which the Crontab triggers. Only in case of the base 

431 type being :class:`str`, parsing occurs. (It's fast and 

432 happens only once for each Crontab instance, so there's no 

433 significant performance overhead involved.) 

434 

435 For the other base types, merely Python type conversions happen. 

436 

437 The argument ``max_`` is needed to determine the expansion of 

438 ``*`` and ranges. The argument ``min_`` is needed to determine 

439 the expansion of ``*`` and ranges for 1-based cronspecs, such as 

440 day of month or month of year. The default is sufficient for minute, 

441 hour, and day of week. 

442 """ 

443 if isinstance(cronspec, int): 

444 result = {cronspec} 

445 elif isinstance(cronspec, str): 

446 result = crontab_parser(max_, min_).parse(cronspec) 

447 elif isinstance(cronspec, set): 447 ↛ 448line 447 didn't jump to line 448, because the condition on line 447 was never true

448 result = cronspec 

449 elif isinstance(cronspec, Iterable): 

450 result = set(cronspec) # type: ignore 

451 else: 

452 raise TypeError(CRON_INVALID_TYPE.format(type=type(cronspec))) 

453 

454 # assure the result does not precede the min or exceed the max 

455 for number in result: 

456 if number >= max_ + min_ or number < min_: 

457 raise ValueError(CRON_PATTERN_INVALID.format( 

458 min=min_, max=max_ - 1 + min_, value=number)) 

459 return result

483 def is_before_last_run(year: int, month: int, day: int) -> bool:

488 def roll_over() -> None: 

489 for _ in range(2000): 

490 flag = (datedata.dom == len(days_of_month) or 

491 day_out_of_range(datedata.year, 

492 months_of_year[datedata.moy], 

493 days_of_month[datedata.dom]) or 

494 (is_before_last_run(datedata.year, 

495 months_of_year[datedata.moy], 

496 days_of_month[datedata.dom]))) 

497 

498 if flag: 

499 datedata.dom = 0 

500 datedata.moy += 1 

501 if datedata.moy == len(months_of_year): 

502 datedata.moy = 0 

503 datedata.year += 1 

504 else: 

505 break 

506 else: 

507 # Tried 2000 times, we're most likely in an infinite loop 

508 raise RuntimeError('unable to rollover, ' 

509 'time specification is probably invalid') 

510 

511 if last_run_at.month in self.month_of_year: 

512 datedata.dom = bisect(days_of_month, last_run_at.day) 

513 datedata.moy = bisect_left(months_of_year, last_run_at.month) 

514 else: 

515 datedata.dom = 0 

516 datedata.moy = bisect(months_of_year, last_run_at.month) 

517 if datedata.moy == len(months_of_year): 

518 datedata.moy = 0 

519 roll_over() 

520 

521 while 1: 

522 th = datetime(year=datedata.year, 

523 month=months_of_year[datedata.moy], 

524 day=days_of_month[datedata.dom]) 

525 if th.isoweekday() % 7 in self.day_of_week: 

526 break 

527 datedata.dom += 1 

528 roll_over() 

529 

530 return ffwd(year=datedata.year, 

531 month=months_of_year[datedata.moy], 

532 day=days_of_month[datedata.dom], 

533 hour=next_hour,

553 def remaining_delta(self, last_run_at: datetime, tz: tzinfo | None = None, 

554 ffwd: type = ffwd) -> tuple[datetime, Any, datetime]: 

555 # caching global ffwd 

556 last_run_at = self.maybe_make_aware(last_run_at) 

557 now = self.maybe_make_aware(self.now()) 

558 dow_num = last_run_at.isoweekday() % 7 # Sunday is day 0, not day 7 

559 

560 execute_this_date = ( 

561 last_run_at.month in self.month_of_year and 

562 last_run_at.day in self.day_of_month and 

563 dow_num in self.day_of_week 

564 ) 

565 

566 execute_this_hour = ( 

567 execute_this_date and 

568 last_run_at.day == now.day and 

569 last_run_at.month == now.month and 

570 last_run_at.year == now.year and 

571 last_run_at.hour in self.hour and 

572 last_run_at.minute < max(self.minute) 

573 ) 

574 

575 if execute_this_hour: 

576 next_minute = min(minute for minute in self.minute 

577 if minute > last_run_at.minute) 

578 delta = ffwd(minute=next_minute, second=0, microsecond=0) 

579 else: 

580 next_minute = min(self.minute) 

581 execute_today = (execute_this_date and 

582 last_run_at.hour < max(self.hour)) 

583 

584 if execute_today: 

585 next_hour = min(hour for hour in self.hour 

586 if hour > last_run_at.hour) 

587 delta = ffwd(hour=next_hour, minute=next_minute, 

588 second=0, microsecond=0) 

589 else: 

590 next_hour = min(self.hour) 

591 all_dom_moy = (self._orig_day_of_month == '*' and 

592 self._orig_month_of_year == '*') 

593 if all_dom_moy: 

594 next_day = min([day for day in self.day_of_week 

595 if day > dow_num] or self.day_of_week) 

596 add_week = next_day == dow_num 

597 

598 delta = ffwd( 

599 weeks=add_week and 1 or 0, 

600 weekday=(next_day - 1) % 7, 

601 hour=next_hour, 

602 minute=next_minute, 

603 second=0, 

604 microsecond=0, 

605 ) 

606 else:

622 def is_due(self, last_run_at: datetime) -> tuple[bool, datetime]: 

623 """Return tuple of ``(is_due, next_time_to_run)``. 

624 

625 If :setting:`beat_cron_starting_deadline` has been specified, the 

626 scheduler will make sure that the `last_run_at` time is within the 

627 deadline. This prevents tasks that could have been run according to 

628 the crontab, but didn't, from running again unexpectedly. 

629 

630 Note: 

631 Next time to run is in seconds. 

632 

633 SeeAlso: 

634 :meth:`celery.schedules.schedule.is_due` for more information. 

635 """ 

636 

637 rem_delta = self.remaining_estimate(last_run_at) 

638 rem_secs = rem_delta.total_seconds() 

639 rem = max(rem_secs, 0) 

640 due = rem == 0 

641 

642 deadline_secs = self.app.conf.beat_cron_starting_deadline 

643 has_passed_deadline = False 

644 if deadline_secs is not None: 

645 # Make sure we're looking at the latest possible feasible run 

646 # date when checking the deadline. 

647 last_date_checked = last_run_at 

648 last_feasible_rem_secs = rem_secs 

649 while rem_secs < 0: 

650 last_date_checked = last_date_checked + abs(rem_delta) 

651 rem_delta = self.remaining_estimate(last_date_checked) 

652 rem_secs = rem_delta.total_seconds() 

653 if rem_secs < 0: 

654 last_feasible_rem_secs = rem_secs 

655 

656 # if rem_secs becomes 0 or positive, second-to-last 

657 # last_date_checked must be the last feasible run date. 

658 # Check if the last feasible date is within the deadline 

659 # for running 

660 has_passed_deadline = -last_feasible_rem_secs > deadline_secs 

661 if has_passed_deadline: 

662 # Should not be due if we've passed the deadline for looking 

663 # at past runs 

664 due = False 

665 

666 if due or has_passed_deadline: 

667 rem_delta = self.remaining_estimate(self.now()) 

668 rem = max(rem_delta.total_seconds(), 0) 

669 return schedstate(due, rem)

809 def remaining_estimate(self, last_run_at: datetime) -> timedelta: 

810 """Return estimate of next time to run. 

811 

812 Returns: 

813 ~datetime.timedelta: when the periodic task should

815 not rise today), returns the time when the next check

819 last_run_at_utc = localize(last_run_at, timezone.utc) 

820 self.cal.date = last_run_at_utc 

821 try: 

822 if self.use_center: 822 ↛ 823line 822 didn't jump to line 823, because the condition on line 822 was never true

823 next_utc = getattr(self.cal, self.method)( 

824 self.ephem.Sun(), 

825 start=last_run_at_utc, use_center=self.use_center 

826 ) 

827 else: 

828 next_utc = getattr(self.cal, self.method)( 

829 self.ephem.Sun(), start=last_run_at_utc 

830 ) 

831 

832 except self.ephem.CircumpolarError: # pragma: no cover 

833 # Sun won't rise/set today. Check again tomorrow

837 timedelta(minutes=1)
