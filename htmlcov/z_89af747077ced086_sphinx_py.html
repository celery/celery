1"""Sphinx documentation plugin used to document tasks. 

2 

3Introduction 

4============ 

5 

6Usage 

7----- 

8 

9The Celery extension for Sphinx requires Sphinx 2.0 or later. 

10 

11Add the extension to your :file:`docs/conf.py` configuration module: 

12 

13.. code-block:: python 

14 

15 extensions = (..., 

16 'celery.contrib.sphinx') 

17 

18If you'd like to change the prefix for tasks in reference documentation 

19then you can change the ``celery_task_prefix`` configuration value: 

20 

21.. code-block:: python 

22 

23 celery_task_prefix = '(task)' # < default 

24 

25With the extension installed `autodoc` will automatically find 

26task decorated objects (e.g. when using the automodule directive) 

27and generate the correct (as well as add a ``(task)`` prefix), 

28and you can also refer to the tasks using `:task:proj.tasks.add` 

29syntax. 

30 

31Use ``.. autotask::`` to alternatively manually document a task. 

32""" 

33from inspect import signature 

34 

35from docutils import nodes 

36from sphinx.domains.python import PyFunction 

37from sphinx.ext.autodoc import FunctionDocumenter 

38 

39from celery.app.task import BaseTask 

40 

41 

42class TaskDocumenter(FunctionDocumenter): 

43 """Document task definitions.""" 

44 

45 objtype = 'task' 

46 member_order = 11 

47 

48 @classmethod 

49 def can_document_member(cls, member, membername, isattr, parent): 

50 return isinstance(member, BaseTask) and getattr(member, '__wrapped__') 

51 

52 def format_args(self): 

53 wrapped = getattr(self.object, '__wrapped__', None) 

54 if wrapped is not None: 

55 sig = signature(wrapped) 

56 if "self" in sig.parameters or "cls" in sig.parameters: 

57 sig = sig.replace(parameters=list(sig.parameters.values())[1:]) 

58 return str(sig) 

59 return '' 

60 

61 def document_members(self, all_members=False): 

62 pass 

63 

64 def check_module(self): 

65 # Normally checks if *self.object* is really defined in the module 

66 # given by *self.modname*. But since functions decorated with the @task 

67 # decorator are instances living in the celery.local, we have to check 

68 # the wrapped function instead. 

69 wrapped = getattr(self.object, '__wrapped__', None) 

70 if wrapped and getattr(wrapped, '__module__') == self.modname: 

71 return True 

72 return super().check_module() 

73

75class TaskDirective(PyFunction): 

76 """Sphinx task directive.""" 

77 

78 def get_signature_prefix(self, sig): 

79 return [nodes.Text(self.env.config.celery_task_prefix)] 

80

82def autodoc_skip_member_handler(app, what, name, obj, skip, options): 

83 """Handler for autodoc-skip-member event.""" 

84 # Celery tasks created with the @task decorator have the property 

85 # that *obj.__doc__* and *obj.__class__.__doc__* are equal, which 

86 # trips up the logic in sphinx.ext.autodoc that is supposed to 

87 # suppress repetition of class documentation in an instance of the 

88 # class. This overrides that behavior. 

89 if isinstance(obj, BaseTask) and getattr(obj, '__wrapped__'): 

90 if skip: 

91 return False 

92 return None 

93 

94 

95def setup(app): 

96 """Setup Sphinx extension.""" 

97 app.setup_extension('sphinx.ext.autodoc') 

98 app.add_autodocumenter(TaskDocumenter) 

99 app.add_directive_to_domain('py', 'task', TaskDirective) 

100 app.add_config_value('celery_task_prefix', '(task)', True) 

101 app.connect('autodoc-skip-member', autodoc_skip_member_handler) 

102 

103 return { 

104 'parallel_read_safe': True 

105 }
