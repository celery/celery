365def flatten(d, root='', keyfilter=_flatten_keys): 

366 """Flatten settings.""" 

367 stack = deque([(root, d)]) 

368 while stack: 

369 ns, options = stack.popleft() 

370 for key, opt in options.items(): 

371 if isinstance(opt, dict): 

372 stack.append((ns + key + '_', opt)) 

373 else: 

374 yield from keyfilter(ns, key, opt)
